14. /**
	 * The utility class for reading/writing csv files.
	 */
	public class CsvUtils
	{
		private static final String CSV_DELIMITER = ",";

		private static final String DOUBLE_QUOTE = "\"";

		private static final String ESCAPED_DOUBLE_QUOTE = "\"\"";

		private static final String SINGLE_QUOTE = "'";

		/**
		 * Builds a line in CSV format based on the list of fields
		 *
		 * @param fields the list of fields
		 * @return the CSV line
		 */
		public static String generateCsvLine(String[] fields)
		{
			return Stream.of(fields).map(CsvUtils::escapeSpecialCharacters).collect(Collectors.joining(CSV_DELIMITER)) + System.lineSeparator();
		}

		/**
		 * Escape the special characters in a CSV field
		 *
		 * @param fieldValue the CSV field value
		 * @return the escaped CVS field
		 */
		private static String escapeSpecialCharacters(String fieldValue)
		{
			String escapedFieldValue = fieldValue;
			// Surround the field with double quotes if it contains the special characters
			if (!StringUtils.isEmpty(fieldValue) && fieldValue.contains(CSV_DELIMITER) || fieldValue.contains(DOUBLE_QUOTE) || fieldValue.contains(SINGLE_QUOTE))
			{
				escapedFieldValue = DOUBLE_QUOTE + fieldValue.replace(DOUBLE_QUOTE, ESCAPED_DOUBLE_QUOTE) + DOUBLE_QUOTE;
			}
			return escapedFieldValue;
		}
	}
15. /**
	 * Creates a zip file for a single file
	 */
	public class SingleFileZipWriter
	{
		private static final long FLUSH_SIZE = 500;

		private ZipOutputStream zipOutputStream;

		private long lineCount = 0;

		/**
		 * Construct
		 *
		 * @param os       the output stream that the zip stream is attached to
		 * @param filename the file that needs to be zipped
		 * @throws IOException the IO exception
		 */
		public SingleFileZipWriter(OutputStream os, String filename) throws IOException
		{
			zipOutputStream = new ZipOutputStream(new BufferedOutputStream(os));
			zipOutputStream.putNextEntry(new ZipEntry(filename));
		}

		/**
		 * Write the content into the zip file line by line
		 *
		 * @param line each line in the file
		 * @throws IOException the IO exception
		 */
		public void writeLine(String line) throws IOException
		{
			if (!StringUtils.isEmpty(line))
			{
				byte bytes[] = line.getBytes("UTF-8");
				zipOutputStream.write(bytes, 0, bytes.length);
				if (lineCount++ % FLUSH_SIZE == 0)
				{
					zipOutputStream.flush();
				}
			}
		}

		/**
		 * Close the Zip stream
		 *
		 * @throws IOException the IO exception
		 */
		public void close() throws IOException
		{
			if (zipOutputStream != null)
			{
				zipOutputStream.flush();
				zipOutputStream.closeEntry();
				zipOutputStream.close();
			}
		}
	}

	private void extractZipToDownloadPath(Response response, String fileName)
    {
        String fileZip = downloadPath.concat(fileName);
        File destDir = new File(downloadPathExtracted);
        byte[] buffer = new byte[response.asByteArray().length];
        ZipEntry zipEntry = null;
        try
        {
            ZipInputStream zis = new ZipInputStream(new FileInputStream(fileZip));
            zipEntry = zis.getNextEntry();
            while (zipEntry != null)
            {
                File newFile = newFile(destDir, zipEntry);
                FileOutputStream fos = new FileOutputStream(newFile);
                int len;
                while ((len = zis.read(buffer)) > 0)
                {
                    fos.write(buffer, 0, len);
                }
                fos.close();
                zipEntry = zis.getNextEntry();
            }
            zis.closeEntry();
            zis.close();
        }
        catch (IOException e)
        {
            e.printStackTrace();
        }
    }

    public static File newFile(File destinationDir, ZipEntry zipEntry) throws IOException
    {
        File destFile = new File(destinationDir, zipEntry.getName());

        String destDirPath = destinationDir.getCanonicalPath();
        String destFilePath = destFile.getCanonicalPath();

        if (!destFilePath.startsWith(destDirPath + File.separator))
        {
            throw new IOException("Entry is outside of the target dir: " + zipEntry.getName());
        }
        return destFile;
    }

	This requires the lambda to have all the dependencies for S3, RDS, and DM
	If the lambda failed, JAMS can retry triggering the lambda in the same day

	this gets rid of all the dirty work the lambda may do, but requires it to have the API running
	a new API endpoint is needed for this approach
	*JAMS Job → run a shell script → Call the API → API does all the work.
	this gets rid of the lambda
	a new endpoint is needed for this approach
	*Spring Scheduler on the API → API does all the work.
	The API needs to be running. If it wasn't, then how would it retry.

1. Adding retry without using any frameork: Need to add a class for Retry. Like below:
	package org.finra.catdd.lambda.retry;

	import java.util.function.Supplier;

	import org.slf4j.Logger;
	import org.slf4j.LoggerFactory;

	/**
	 * Class with retry methods to external call from lambda
	 *
	 * @param <T>
	 */
	public class Retry<T>
	{
		private static final Logger LOGGER = LoggerFactory.getLogger(Retry.class);

		private int maxRetries;

		private int retryCounter;

		public Retry(int maxRetries)
		{
			this.maxRetries = maxRetries;
		}

		// Takes a function and executes it, if fails, passes the function to the retry command
		public T run(Supplier<T> function)
		{
			try
			{
				return function.get();
			}
			catch (Exception e)
			{
				return retry(function);
			}
		}

		public int getRetryCounter()
		{
			return retryCounter;
		}

		private T retry(Supplier<T> function) throws RuntimeException
		{
			LOGGER.error("FAILED - call failed, will be retried " + maxRetries + " times.");
			retryCounter = 1;
			while (retryCounter <= maxRetries)
			{
				try
				{
					Thread.sleep(2000 * retryCounter);
					return function.get();
				}
				catch (Exception ex)
				{
					LOGGER.error("FAILED - failed on retry " + retryCounter + " of " + maxRetries + ", error: " + ex);
					if (retryCounter >= maxRetries)
					{
						LOGGER.error("Max retries exceeded.");
						break;
					}
					retryCounter++;
				}
			}
			throw new RuntimeException("Call failed on all of " + maxRetries + " retries");
		}
	}

		 * Generic Method that generates and returns a Cache Configuration
		 *
		 * @param maxEntriesHeap maximum entries the heap can have
		 * @param ttl the time to live duration in seconds
		 *
		 * @return the Cache Configuration
		 */
		private javax.cache.configuration.Configuration<Object, Object> createConfiguration(long maxEntriesHeap, Duration ttl)
		{
			LOGGER.info("maxEntriesHeap = " + maxEntriesHeap + ", timeToLiveExpiration = " + ttl.getSeconds());
			return Eh107Configuration.fromEhcacheCacheConfiguration(
				CacheConfigurationBuilder.newCacheConfigurationBuilder(Object.class, Object.class,
					ResourcePoolsBuilder.heap(maxEntriesHeap))
					.withExpiry(ExpiryPolicyBuilder.timeToLiveExpiration(ttl))

				 return true;

	@PostMapping(value = "/atsOrderTypes/{crtd_by}")
    @ResponseStatus(code = HttpStatus.CREATED)
    public Map<String, Object> createAtsOrderType(@Valid @RequestBody AtsOrderType new_atsOrderType, BindingResult bindingResult,
        @PathVariable("crtd_by") String crtd_by)
    {
        LOGGER.info(" \\nPOST ATS invoked -> ");
        Map<String, Object> map = new HashMap<>();
        if (bindingResult.hasErrors())
        {
            map.put("Status", "Failed to create Oso Relationship");
            Map<String, String> errMsgMap = new HashMap<String, String>();
            List<ObjectError> errors = bindingResult.getAllErrors();
            for (ObjectError objectError : errors)
            {
                if (objectError instanceof FieldError)
                {
                    String fieldError = ((FieldError) objectError).getField();
                    errMsgMap.put(fieldError, objectError.getDefaultMessage());
                }
                else if (objectError instanceof ObjectError)
                {
                    LOGGER.info(objectError);
                    errMsgMap.put(objectError.getCode(), objectError.getDefaultMessage());
                }
            }
            map.put("Errors", errMsgMap);
            LOGGER.error(errMsgMap);
            return map;
        }
        else
        {
            return atsOrderTypeService.createAtsOrderType(new_atsOrderType, crtd_by);
        }
    }
